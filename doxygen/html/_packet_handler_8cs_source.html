65 public override Packet Handle(Packet packet)
66 {
67 try
68 {
69 if (resultWriter == null)
70 throw new Exception("_textWriter is null");
71 if (encoding == null)
72 throw new Exception("encoding is null");
73
74#warning Finish the Handle method
75
76 //_responses[packet.Flags]
77
78 switch ((PacketFlags.Flags)packet.FlagsByte)
79 {
80 //=============
81 //Responses
82 case PacketFlags.Flags.Response:
83 case PacketFlags.Flags.RspSingleMsg:
84 resultWriter.WriteLine(packet.ToString());
85
86 break;
87
88
89 //=============
90 //Messages
91 case PacketFlags.Flags.SingleMsg:
92 //Clear operation
93 resultWriter.WriteLine(encoding.GetString(packet.Bytes));
94
95 break;
96 case PacketFlags.Flags.StartMsg:
97 longBuffer.Add(packet.Bytes);
98
99 break;
100
101 case PacketFlags.Flags.EndMsg:
102 //Clear operation
103 longBuffer.Add(packet.Bytes);
104
105 foreach (byte[] bytes in longBuffer)
106 resultWriter.Write(encoding.GetString(bytes));
107 resultWriter.WriteLine();
108
109 longBuffer.Clear();
110
111 break;
112
113 // If the _longBuffer is in use, then the
114 // base messages are added to it
115 case PacketFlags.Flags.Message:
116 if (longBuffer.Count > 0)
117 longBuffer.Add(packet.Bytes);
118
119 break;
120
121 //=============
122 //Files
123 //https://code-maze.com/convert-byte-array-to-file-csharp/
124 case PacketFlags.Flags.File | PacketFlags.Flags.Single:
125 FileStruct fstruct = FileStruct.GetStruct(packet);
126 if (fstruct.Name.Length == 0)
127 fstruct.Name = Path.GetRandomFileName();
128 if (File.Exists(fstruct.Name))
129 fileStream = new FileStream(fstruct.Name, FileMode.Truncate);
130 else
131 fileStream = new FileStream(fstruct.Name, FileMode.CreateNew);
132 fileStream.Write(fstruct.Data, 0, fstruct.Data.Length);
133 fileStream.Close();
134
135 resultWriter.WriteLine($"File {fstruct.Name} recieved");
136 break;
137
138 //Clear operation
139 case PacketFlags.Flags.File | PacketFlags.Flags.Start:
140 throw new NotImplementedException();
141 case PacketFlags.Flags.File | PacketFlags.Flags.End:
142 //Clear operation
143 throw new NotImplementedException();
144 case PacketFlags.Flags.File:
145 throw new NotImplementedException();
146
147 case (PacketFlags.Flags)0b_0000_1000:
148 throw new NotImplementedException();
149
150 case PacketFlags.Flags.None:
151 throw new ArgumentNullException("Packet's has no flags");
152 }
153
154 }
155 catch (Exception e)
156 {
157 return new Packet(
158 (byte)PacketFlags.Flags.RspSingleErr,
159 e.Message + " | {" + packet + "}",
160 packet.Id);
161 }
162
163 return new Packet(
164 (byte)PacketFlags.Flags.RspSingleMsg,
165 _sResponses_[(PacketFlags.Flags)packet.FlagsByte] + ":: {" + packet + "}",
166 packet.Id);
167 }
168 }
